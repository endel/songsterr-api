def camelize(first_letter = :upper)
  parts = self.split('_')
  i = 0
  parts.collect! do |p|
    p = p.capitalize if (i != 0 || (i == 0 && first_letter == :upper))
    i += 1
    p
  end
  parts.join('')
end
